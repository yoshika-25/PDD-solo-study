Advanced Python Interview Quiz

1. What's the output of this complex decorator?


def counter(func):
    def wrapper(*args, **kwargs):
        wrapper.calls += 1
        return func(*args, **kwargs)
    wrapper.calls = 0
    return wrapper

@counter
def example(): pass

example()
example()
print(example.calls)

2. What will this generator produce?


def custom_generator():
    x = 1
    while True:
        yield x
        x = x * 2 + 1

gen = custom_generator()
print(next(gen), next(gen), next(gen))

3. What's the output of this closure?


def multiplier(x):
    def inner(y):
        return x * y
    return inner

fn = multiplier(3)
print(fn(4))

4. What happens with this list comprehension?


result = [x for x in range(10) if x % 2 == 0]
print(result)

5. What is printed by this code?


class MyClass:
    class_var = 10

    def __init__(self, val):
        self.instance_var = val

obj1 = MyClass(5)
obj2 = MyClass(8)

print(obj1.class_var, obj2.class_var)

6. What's the output of this map function?


def square(x):
    return x ** 2

numbers = [1, 2, 3, 4]
result = list(map(square, numbers))
print(result)

7. What does this lambda function do?


fn = lambda x, y: x if x > y else y
print(fn(10, 5))

8. What is the output?


def outer():
    x = 10
    def inner():
        nonlocal x
        x += 5
        return x
    return inner

fn = outer()
print(fn(), fn())
